📋 Guía de Integración Real

1. Instagram Graph API

Pasos:
1. Crear app en Facebook Developers
2. Configurar Instagram Basic Display o Graph API
3. Obtener Access Token (requiere Business Verification)
4. Usar endpoint: POST /{ig-user-id}/media

Costo: Gratis (límites aplican)
Docs: developers.facebook.com/docs/instagram-api

2. Facebook Graph API

Pasos:
1. Misma app de Facebook Developers
2. Solicitar permisos: pages_manage_posts, pages_read_engagement
3. Obtener Page Access Token
4. Endpoint: POST /{page-id}/feed

Costo: Gratis
Docs: developers.facebook.com/docs/graph-api

3. WhatsApp Business API

Opción A - Meta (Oficial):
• Requiere Meta Business Verification
• Costo: ~$0.005-0.04 por mensaje
• Complejo de implementar

Opción B - Twilio (Recomendado):
• Más fácil de implementar
• Costo: ~$0.005-0.02 por mensaje
• Sandbox disponible para pruebas

Opción C - UltraMsg (Simple):
• HTTP API simple
• Costo: Desde $9/mes
• Ideal para empezar rápido

Para empezar AHORA (gratis):
Usar: wa.me links (como está implementado)
Ejemplo: https://wa.me/56966874954?text=Hola

// ============================================
// INTEGRACIÓN INSTAGRAM GRAPH API (Node.js)
// ============================================

const axios = require('axios');

class InstagramAPI {
    constructor(accessToken, instagramAccountId) {
        this.accessToken = accessToken;
        this.instagramAccountId = instagramAccountId;
        this.baseUrl = 'https://graph.facebook.com/v18.0';
    }

    // Subir imagen (Paso 1)
    async uploadImage(imageUrl, caption) {
        try {
            const response = await axios.post(
                `${this.baseUrl}/${this.instagramAccountId}/media`,
                {
                    image_url: imageUrl,
                    caption: caption,
                    access_token: this.accessToken
                }
            );
            return response.data.id; // Media ID
        } catch (error) {
            console.error('Error subiendo imagen:', error);
            throw error;
        }
    }

    // Publicar (Paso 2)
    async publish(mediaId) {
        try {
            const response = await axios.post(
                `${this.baseUrl}/${this.instagramAccountId}/media_publish`,
                {
                    creation_id: mediaId,
                    access_token: this.accessToken
                }
            );
            return response.data;
        } catch (error) {
            console.error('Error publicando:', error);
            throw error;
        }
    }

    // Programar publicación (usando cron job)
    async schedulePost(imageUrl, caption, scheduledTime) {
        // Guardar en base de datos
        const post = await db.posts.create({
            imageUrl,
            caption,
            scheduledTime,
            status: 'pending'
        });

        // El cron job revisa cada minuto y publica
        return post;
    }
}

const twilio = require('twilio');

class WhatsAppService {
    constructor(accountSid, authToken) {
        this.client = twilio(accountSid, authToken);
        this.fromNumber = 'whatsapp:+14155238886'; // Número de Twilio
        this.toNumber = 'whatsapp:+56966874954'; // Tu número
    }

    async sendReminder(message) {
        try {
            const response = await this.client.messages.create({
                from: this.fromNumber,
                to: this.toNumber,
                body: message
            });
            console.log('Mensaje enviado:', response.sid);
            return response;
        } catch (error) {
            console.error('Error enviando WhatsApp:', error);
            throw error;
        }
    }

    // Enviar recordatorio de post programado
    async sendPostReminder(post) {
        const message = `📅 *Recordatorio Kany Manager*
        
Tienes un post programado:
🕐 ${post.scheduledTime}
📱 ${post.platform}
📝 ${post.caption.substring(0, 50)}...

¡Es hora de publicar!`;
        
        return this.sendReminder(message);
    }
}

// INTEGRACIÓN WHATSAPP (UltraMsg - Simple)
// ============================================

class UltraMsgService {
    constructor(instanceId, token) {
        this.instanceId = instanceId;
        this.token = token;
        this.baseUrl = `https://api.ultramsg.com/${instanceId}`;
    }

    async sendMessage(to, message) {
        try {
            const response = await axios.post(`${this.baseUrl}/messages/chat`, {
                token: this.token,
                to: to,
                body: message,
                priority: 10
            });
            return response.data;
        } catch (error) {
            console.error('Error:', error);
            throw error;
        }
    }

    async sendImage(to, imageUrl, caption) {
        try {
            const response = await axios.post(`${this.baseUrl}/messages/image`, {
                token: this.token,
                to: to,
                image: imageUrl,
                caption: caption
            });
            return response.data;
        } catch (error) {
            console.error('Error:', error);
            throw error;
        }
    }
}

// Ejemplo: Programar post con recordatorio WhatsApp
async function schedulePostWithReminder(postData) {
    // 1. Guardar en base de datos
    const post = await db.posts.create(postData);
    
    // 2. Programar publicación Instagram
    await instagramAPI.schedulePost(
        postData.imageUrl,
        postData.caption,
        postData.scheduledTime
    );
    
    // 3. Enviar recordatorio WhatsApp (1 hora antes)
    const reminderTime = new Date(postData.scheduledTime);
    reminderTime.setHours(reminderTime.getHours() - 1);
    
    setTimeout(() => {
        whatsAppService.sendPostReminder(post);
    }, reminderTime - new Date());
    
    return post;
}
